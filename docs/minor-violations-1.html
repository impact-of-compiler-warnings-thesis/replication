Chapter 10 Minor Violations

This section covers the analysis relating to the “minor violations” metric.

10.1 Prepare data

# The model seed is the date that the analysis was conducted
model_seed <- 20230328

create_minor_violations_data <- function(data) {
  list(
    minor_violations_kloc_std=scale(data$minor_violations_kloc),
             contributors_std=scale(data$contributors),
                    stars_std=scale(data$stars),
                    files_std=scale(data$files),
                     size_std=scale(data$loc),
                      age_std=scale(get_age(data$created_at))
  )
}

# Prepare data with all categories as factors (1-11).
data <- create_minor_violations_data(df)
data$category <- factor(df$category)

# Prepare data that factors on the presence of compiler warnings
data_factor_uses_warnings <- create_minor_violations_data(df)
data_factor_uses_warnings$category <- factor(uses_warnings(df$category))

10.2 Define models

10.2.1 MIV1

fit_model_miv1 <- function(name, data) {
  fit_model(name,
            "minor_violations_kloc_std ~ 0 + category",
            c(
              prior(normal(0, 1), class=b),
              prior(exponential(1), class=sigma)
            ),
            data,
            model_seed)
}

miv1 <- fit_model_miv1("miv1", data)
